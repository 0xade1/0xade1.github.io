ELF = "./patched-be-quick-or-be-dead-1"
ADDRESS=0x0040070a

INSTRUCTIONS = "mov eax, 0xeb866516 ; pop rbp ; ret"

def patch_instructions(address, instructions):
    r2 = r2pipe.open(ELF)
    r2.cmd(f"s {address}")
    r2.cmd("oo+")
    r2.cmd(f'"wa {instructions}"')
    r2.cmd("q")

patch_instructions(ADDRESS, INSTRUCTIONS)
os.system(ELF)

We can alternatively calculate ourselves the fibonacci number if we respect two rules:

  1. keep only first 32bits to simulate the dword overflow
  2. avoid recursion to avoid RuntimeError: maximum recursion depth exceeded

Solution:

  1. We can simulate the overflow by using numpy.uint32(..) or number & 0xFFFFFFFF
  2. To avoid recursion, we can use a Memoization decorator that will cache in a dictionary any previous fibonacci’s number it calculated in the past

#!/usr/bin/env python
import numpy as np

np.seterr(all="ignore")				#ignore warnings due tu overflows

class Memoize(object):
    def __init__(self, func):
        self.func = func
        self.cache = {}
    def __call__(self, *args):
        if args in self.cache:
            return self.cache[args]
        ret = self.func(*args)
        self.cache[args] = ret
        return ret

@Memoize
def fib(n):
    if n == 0:
        return np.uint32(0)
    elif n == 1:
        return np.uint32(1)
    return fib(n-2) + fib(n-1)


if __name__ == "__main__":
    requested_number = 1083
    for i in range(requested_number-1):		#cache 1082 fibonacci's number
        fib(i)
    print(hex(fib(requested_number)))

def calc(n):
  if n <= 4:
    return n ** 2 + 0x2345
  else:
    ebx = calc(n-1) - calc(n-2)
    r12d = calc(n-3) - calc(n-4)
    ebx += r12d
    eax = calc(n-5)
    return eax * 0x1234 + ebx

If we execute the above python code calc(0x1865b), we will have a RuntimeError: maximum recursion depth exceeded while getting the repr of an object, so we need few adjustments to turn the recursive algorithm into iteractive execution:

  1. As we did with large fibonacci’s number, we can cache all results using a function or class decorator.
  2. We also need to simulate the overflow of very large number, using numpy.int32(...)

#!/usr/bin/env python

import numpy as np
import functools
import sys
import ctypes

np.seterr(all="ignore")

def memoize(func):
    cache = func.cache = {}

    @functools.wraps(func)
    def memoized_func(*args, **kwargs):
        key = str(args) + str(kwargs)
        if key not in cache:

def find_return_address(asm):			#Read the disassembled code of a function and return the __ret__ instruction address
    for line in asm.split('\n'):
        if 'ret' in line:
            index = line.find('0x')
            return line[index:index+10]

def debug():
    r2 = r2pipe.open(BINARY)
    r2.cmd('aa')                                    #Analyze symbols
    asm = r2.cmd('pdf @ %s' % ADDRESS)              #store the disassembled code in asm
    r2.cmd('ood')                                   #enable debug mode

    r2.cmd('dr rip=%s' % find_return_address(asm))  #change rip to point to 'ret' instruction
    return r2.cmd('dc')                             #continue the programme execution

The rol and ror instructions are similar to shl and shr except the shifted bits are rotated to the other end.

Also, the rotation is on AL which is 1 byte size; therefore the rotation is on 8 bits

The movzx instruction at address 0x080485f2 loads unsigned 8-bit value into 32-bit register eax and zero-extend

def encrypt(plaintext):
    ciphertext = ''
    for i in range(len(plaintext)):
        ch = ord(plaintext[i])		#we want the int representation of the char
        ch = rol(ch, 4, word_size = 8)	#left shift on a Byte (size=8) see sym.rol4 in radare2
        ch = ch ^ 0x16
        ch = ror(ch, 8, word_size = 8)  #right shift on a Byte (size=8) see sym.ror8 in radare2
        ciphertext += chr(ch)
    return ciphertext

def decrypt(ciphertext):
    plaintext = ''
    for i in range(len(ciphertext)):
        ch = ord(ciphertext[i])
        ch = rol(ch, 8, word_size = 8)
        ch ^= 0x16
        ch = ror(ch, 4, word_size = 8)
        plaintext += chr(ch)
    return plaintext
